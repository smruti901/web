Pyramid Program

                          
                            public static void main(String[] args) {
                              int rows = 5; // You can change the number of rows here
                              
                              for (int i = 1; i <= rows; i++) {
                                  // Print spaces before the asterisks
                                  for (int j = 1; j <= rows - i; j++) {
                                      System.out.print(" ");
                                  }
                                  
                                  // Print asterisks
                                  for (int k = 1; k <= 2 * i - 1; k++) {
                                      System.out.print("*");
                                  }
                                  
                                  // Move to the next line for the next row
                                  System.out.println();
                              }
                          }

Diamond Program

                          
                            public static void main(String[] args) {
                              int n = 5; // You can change the number of rows here
                      
                              // Upper half of the diamond
                              for (int i = 1; i <= n; i++) {
                                  // Print spaces before the asterisks
                                  for (int j = 1; j <= n - i; j++) {
                                      System.out.print(" ");
                                  }
                      
                                  // Print asterisks
                                  for (int k = 1; k <= 2 * i - 1; k++) {
                                      System.out.print("*");
                                  }
                      
                                  // Move to the next line for the next row
                                  System.out.println();
                              }
                      
                              // Lower half of the diamond
                              for (int i = n - 1; i >= 1; i--) {
                                  // Print spaces before the asterisks
                                  for (int j = 1; j <= n - i; j++) {
                                      System.out.print(" ");
                                  }
                      
                                  // Print asterisks
                                  for (int k = 1; k <= 2 * i - 1; k++) {
                                      System.out.print("*");
                                  }
                      
                                  // Move to the next line for the next row
                                  System.out.println();
                              }
                          }

Right Justified Triangle Mirror Program

                          
                            public static void main(String[] args) {
                              int n = 5; // You can change the number of rows here
                      
                              // Right triangle (right-justified)
                              for (int i = 1; i <= n - 1; i++) {
                                  // Print spaces before the asterisks
                                  for (int j = 1; j <= n - i; j++) {
                                      System.out.print(" ");
                                  }
                      
                                  // Print asterisks
                                  for (int k = 1; k <= i; k++) {
                                      System.out.print("*");
                                  }
                      
                                  // Move to the next line for the next row
                                  System.out.println();
                              }
                      
                              // Common line for merging the triangles
                              for (int i = 1; i <= n; i++) {
                                  System.out.print("*");

Sort String Using Array

                          
                            import java.lang.*;
                            import java.io.*;
                            import java.util.*;
                            import java.util.Arrays;

                            public class Test {
                                public static void main(String[] args) {
                                    String input = "hello";
                                    String sortedString = sortString(input);
                                    System.out.println("Original string: " + input);
                                    System.out.println("Sorted string: " + sortedString);
                                }

                                public static String sortString(String str) {                                    
                                    char[] charArray = str.toCharArray();
                                    Arrays.sort(charArray);
                                    return new String(charArray);
                                }
                            }

Sort a String Alpabetically Using Array

                          
                            import java.lang.*;
                            import java.io.*;
                            import java.util.*;
                            import java.util.Arrays;

                            public class Test {
                                public static void main(String[] args) {
                                    String sentence = "Example to Sort"; 
                                    String[] words = sentence.split(" ");

                                    for (int i = 0; i < words.length; i++) {
                                        words[i] = words[i].toLowerCase();
                                    }
                                    
                                    Arrays.sort(words);
                                    
                                    String sortedSentence = String.join(" ", words);
                                    
                                    System.out.println("Original: " + sentence);
                                    System.out.println("Sorted: " + sortedSentence);
                                }
                            }

                            Original: Example to Sort
                            Sorted: example sort to
