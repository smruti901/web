Largest element in an array

                          
                            import java.lang.*;
                            import java.io.*;
                            import java.util.*;
                            
                            public class Test {
                               public static int findMaxElement(int[] arr) {
                                      int max = arr[0];
                                      for (int num : arr) {
                                          if (num > max) {
                                              max = num;
                                          }
                                      }
                                      return max;
                                  }
                            
                                  public static void main(String[] args) {
                                      int[] array = { 10, 5, 20, 8, 15 };
                                      int maxElement = findMaxElement(array);
                                      System.out.println("Maximum element: " + maxElement);
                                  }
                            }

Get array input Using Scanner Class

                          
                            import java.lang.*;
                            import java.io.*;
                            import java.util.*;

                            public class Test {
                              public static void main(String[] args) {
                                      Scanner scanner = new Scanner(System.in);
                                      
                                      System.out.print("Enter the length of the array: ");
                                      int length = scanner.nextInt();
                                      
                                      int[] array = new int[length];
                                      
                                      System.out.println("Enter the elements of the array:");
                                      for (int i = 0; i < length; i++) {
                                          array[i] = scanner.nextInt();
                                      }
                                      
                                      // Print the entered array
                                      System.out.println("Entered array:");
                                      for (int num : array) {
                                          System.out.print(num + " ");
                                      }
                                      
                                      // Remember to close the scanner
                                      scanner.close();
                                  }
                            }

Merge and Sort Array using Bubble Sort

                          
                            import java.lang.*;
                            import java.io.*;
                            import java.util.*;

                            public class Test {
                              public static void main(String[] args) {
                                  int[] array1 = { 15, 10, 200 };
                                  int[] array2 = { 81, 15, 30 };

                                  int length1 = array1.length;
                                  int length2 = array2.length;
                                  int[] mergedArray = new int[length1 + length2];

                                  int i = 0, j = 0, k = 0;

                                  while (i < length1 && j < length2) {
                                      mergedArray[k++] = array1[i++];
                                      mergedArray[k++] = array2[j++];
                                  }

                                  while (i < length1) {
                                      mergedArray[k++] = array1[i++];
                                  }

                                  while (j < length2) {
                                      mergedArray[k++] = array2[j++];
                                  }

                                  System.out.println("Merged array:");
                                  for (int num : mergedArray) {
                                      System.out.print(num + " ");
                                  }
                                  
                                  System.out.println();  
                                  
                                  for (i = 0; i < mergedArray.length - 1; i++) {
                                      for (j = 0; j < mergedArray.length - i - 1; j++) {
                                          if (mergedArray[j] > mergedArray[j + 1]) {
                                              int temp = mergedArray[j];
                                              mergedArray[j] = mergedArray[j + 1];
                                              mergedArray[j + 1] = temp;
                                          }
                                      }
                                  }
                                
                                  System.out.println("Sorted array:");
                                  for (int num : mergedArray) {
                                      System.out.print(num + " ");
                                  }
                              }
                            }

O/P

                            15 81 10 15 200 30 
                            Sorted array:

Sort String Using Array

                          
                            import java.lang.*;
                            import java.io.*;
                            import java.util.*;
                            import java.util.Arrays;

                            public class Test {
                                public static void main(String[] args) {
                                    String input = "hello";
                                    String sortedString = sortString(input);
                                    System.out.println("Original string: " + input);
                                    System.out.println("Sorted string: " + sortedString);
                                }

                                public static String sortString(String str) {                                    
                                    char[] charArray = str.toCharArray();
                                    Arrays.sort(charArray);
                                    return new String(charArray);
                                }
                            }

O/P

                          
                            Original string: hello
                            Sorted string: ehllo

Sort a String Alpabetically Using Array

                            import java.util.*;
                            import java.util.Arrays;

                            public class Test {
                                public static void main(String[] args) {
                                    String sentence = "Example to Sort"; 
                                    String[] words = sentence.split(" ");

                                    for (int i = 0; i < words.length; i++) {
                                        words[i] = words[i].toLowerCase();

                                    String sortedSentence = String.join(" ", words);

                            }
