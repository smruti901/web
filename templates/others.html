                              String s ="my name is java";

                              String capitalizeword = "";
                              for (String word : words) {
                                String first = word.substring(0,1);
                                String afterfirst = word.substring(1);

                              for (int i = 0; i < str.length(); i++) {
                                  char currentChar = Character.toLowerCase(str.charAt(i));
                                  if (currentChar != ' ' && currentChar >= 'a' && currentChar <= 'z' &&
                                      currentChar != 'a' && currentChar != 'e' && currentChar != 'i' && 
                                      currentChar != 'o' && currentChar != 'u') {
                                      count++;
                                  }
                              }
                              
                              System.out.print("Number of consonants: " + count);
                          }

Count vowels using Map

                          
                            import java.util.*;
                            public static void main(String[] args) {
                              String str = "Hello, World!";
                              Map map = new HashMap<>();
                              map.put("a", 0);
                              map.put("e", 0);
                              map.put("i", 0);
                              map.put("o", 0);
                              map.put("u", 0);
                              for (int i = 0; i < str.length(); i++) {
                                char ch = str.charAt(i);
                                if(ch == 'a' || ch == 'A') {
                                  map.put("a", map.get("a")+1);
                                }
                                else if(ch == 'e' || ch == 'E') {
                                  map.put("e", map.get("e")+1);
                                }
                                else if(ch == 'i' || ch == 'I') {
                                  map.put("i", map.get("1")+1);
                                }
                                else if(ch == 'o' || ch == 'O') {
                                  map.put("o", map.get("o")+1);
                                }
                                else if(ch == 'u' || ch == 'U') {
                                  map.put("u", map.get("u")+1);
                                }
                              }
                              System.out.println("count of 'a' ="+ map.get("a"));
                              System.out.println("count of 'e' ="+ map.get("e"));
                              System.out.println("count of 'i' ="+ map.get("i"));
                              System.out.println("count of 'o' ="+ map.get("o"));
                              System.out.println("count of 'u' ="+ map.get("u"));
                              
                            }

Fibonacci Series

                          
                            import java.util.*;
                            public static void main(String[] args) {
                              int prev=0;
                              int next=1;
                              int range=5;
                              System.out.print(prev+" ");
                              System.out.print(next+" ");
                              for (int i = 1; i <= range; i++) {
                                int sum=prev+next;
                                System.out.print(sum+" ");
                                prev=next;
                                next=sum;
                          
                              }
                            }
